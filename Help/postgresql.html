          TABLESPACE = pg_default

          CONNECTION LIMIT = -1;

SQL - Create Table

Se muestra ejemplo de Instancia en este caso para la clase con acceso a la Base de Datos 

      
  Create table Manufactura_celda
  (
    id_cel Integer NOT NULL,
    id_aer Integer NOT NULL,
    id_man_f Integer NOT NULL,
    id_man_c Integer NOT NULL,
    realizada boolean DEFAULT false,
    avance Integer DEFAULT 0,           --0 - 100 %
    observaciones character varying(10000),
    Status Integer DEFAULT 0,           --0: Eliminado, 1: No Eliminado
    HoraUpdate time without time zone,--Hora de alta
    FechaUpdate Date,         --Fecha de alta
    Hora time without time zone,--Hora de alta
    Fecha Date,         --Fecha de alta
    primary key (id_cel)
  ) With Oids;
