Gridview - ItemTemplate

Se validan controles Enabled y Visible usando código C# 

      
  <ItemTemplate>
    <asp:DropDownList runat="server" ID="cbStatus" AppendDataBoundItems="true" Width="100px" Enabled='<%# IsVisible() %>'> </asp:DropDownList>
    <asp:Button runat="server" ID="btActualizar" Text="Actualizar" CommandName="EnAlmacenUpdate" CommandArgument='<%# Eval("id_usu") %>' Visible='<%# IsVisible() %>' Font-Size="10px" Width="60px"></asp:Button>
    <br />
  </ItemTemplate>
      
      
  public bool IsVisible()
  {
      if (DataAccess.Instancia().UsuarioVerNivel(SessionState.Current.User) == (int)TipoUsuario.Administrador)
      {
          return true;
      }
      else
      {
          return false;
      }
  }
